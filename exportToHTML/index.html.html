    <script>
        let progress = 100;
        let isUnlocked = false;
        const circle = document.getElementById('circleProgress');
        const circumference = 2 * Math.PI * 100;
        circle.style.strokeDasharray = circumference;

        const audioCtx = new (window.AudioContext || window.webkitAudioContext)();

        function playSparkle(freq, delay, vol = 0.4) {
            const osc = audioCtx.createOscillator();
            const gain = audioCtx.createGain();
            osc.type = 'sine';
            osc.frequency.setValueAtTime(freq, audioCtx.currentTime + delay);
            gain.gain.setValueAtTime(0, audioCtx.currentTime + delay);
            gain.gain.linearRampToValueAtTime(vol, audioCtx.currentTime + delay + 0.05);
            gain.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + delay + 0.5);
            osc.connect(gain);
            gain.connect(audioCtx.destination);
            osc.start(audioCtx.currentTime + delay);
            osc.stop(audioCtx.currentTime + delay + 0.5);
        }

        const content = {

        function handleTap() {
            if (isUnlocked) return;
            if (audioCtx.state === 'suspended') audioCtx.resume();

            document.getElementById('lobbyMusic').play().catch(e => {});

            const osc = audioCtx.createOscillator();
            const g = audioCtx.createGain();
            osc.frequency.value = 400 + (100 - progress);
            g.gain.setValueAtTime(0.6, audioCtx.currentTime);
            g.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + 0.15);
            osc.connect(g);
            g.connect(audioCtx.destination);
            osc.start();
            osc.stop(audioCtx.currentTime + 0.15);

            if (navigator.vibrate) navigator.vibrate(15);

            progress -= 10;
            if (progress < 0) progress = 0;
            const fillProgress = (100 - progress) / 100;
            document.getElementById('fill').style.clipPath = `inset(${progress}% 0 0 0)`;
            circle.style.strokeDashoffset = circumference - (fillProgress * circumference);
            if (progress <= 0) unlock();
        }

        async function typeEffect(element, text, speed = 40) {
            for (let i = 0; i < text.length; i++) {
                element.innerHTML += text.charAt(i);
                const area = document.getElementById('scrollArea');
                area.scrollTo({ top: area.scrollHeight, behavior: 'smooth' });

                let currentDelay = speed;
                const typedSoFar = text.substring(0, i + 1);
                if (typedSoFar.endsWith("heart tal¬≤")) { currentDelay = 1300; }
                else if (text.charAt(i) === '.') { currentDelay = 450; }
                await new Promise(r => setTimeout(r, currentDelay));
            }
        }

        async function unlock() {
            isUnlocked = true;
            [880, 1108, 1318, 1760].forEach((f, i) => { playSparkle(f, i * 0.1); });

            document.getElementById('gameUI').style.display = 'none';
            document.getElementById('secretContent').style.display = 'block';
            confetti({ particleCount: 150, spread: 80, origin: { y: 0.6 } });
            createPetals();

            await new Promise(r => setTimeout(r, 1200));
            await typeEffect(document.getElementById('mainTitle'), content.title, 60);
            await new Promise(r => setTimeout(r, 1500));

            for (const text of content.paragraphs) {
                const p = document.createElement('p');
                p.className = 'reveal-text';
                document.getElementById('messageBody').appendChild(p);
                await typeEffect(p, text, 40);
                await new Promise(r => setTimeout(r, 600));
            }

            document.getElementById('footerArea').style.display = 'block';
            await typeEffect(document.getElementById('flowerLabel'), "A little flower to brighten your day -harolddy", 40);

            setTimeout(() => {
                document.getElementById('flowerCont').style.opacity = "1";
                const stem = document.getElementById('flowerStem');
                stem.classList.add('growing-stem', 'show-leaves');
                document.getElementById('flowerHead').classList.add('blooming-head');
                document.getElementById('glowEffect').classList.add('active');

                setTimeout(() => { document.getElementById('swayCont').classList.add('is-dancing'); }, 2500);
                setTimeout(() => document.getElementById('scrollArea').scrollTo({ top: document.getElementById('scrollArea').scrollHeight, behavior: 'smooth' }), 500);
                setTimeout(() => {
                    document.getElementById('nextBtnContainer').style.display = 'block';
                    document.getElementById('scrollArea').scrollTo({ top: document.getElementById('scrollArea').scrollHeight, behavior: 'smooth' });
                }, 4000);
            }, 500);
        }

        async function showNextSlide() {
            if (audioCtx.state === 'suspended') audioCtx.resume();
            playSparkle(1200, 0, 0.5);
            if (navigator.vibrate) navigator.vibrate(20);

            document.getElementById('slide1').style.display = 'none';
            document.getElementById('nextBtnContainer').style.display = 'none';
            document.getElementById('slide2').style.display = 'block';
            document.getElementById('scrollArea').scrollTop = 0;

            await typeEffect(document.getElementById('slide2Title'), slide2Content.title, 60);
            await new Promise(r => setTimeout(r, 1000));

            for (const text of slide2Content.paragraphs) {
                const p = document.createElement('p');
                p.className = 'reveal-text';
                if(text.includes("cringey-an")) p.style.fontStyle = "italic";
                if(text.includes("Happy Valentine's")) {
                    p.style.fontWeight = "600"; p.style.color = "var(--heart-full)"; p.style.textAlign = "center";
                }
                document.getElementById('slide2Body').appendChild(p);
                await typeEffect(p, text, 40);
                await new Promise(r => setTimeout(r, 600));
            }
            confetti({ particleCount: 200, spread: 100, origin: { y: 0.8 } });
        }

        function createPetals() {
            setInterval(() => {
                const p = document.createElement('div');
                p.className = 'petal';
                const s = Math.random() * 15 + 10 + 'px';
                p.style.width = s; p.style.height = s;
                p.style.left = Math.random() * 100 + 'vw';
                p.style.top = '-5vh';
                p.style.animation = `petalFall ${Math.random() * 5 + 5}s linear forwards`;
                document.body.appendChild(p);
                setTimeout(() => p.remove(), 8000);
            }, 500);
        }
